6. JSON Student Management (Program 1)
Start the program and import json module for file handling.
Define filename where student records will be stored permanently.
Create function to load data from JSON file.
Handle file not found exception by returning empty list.
Create function to save updated data into JSON file.
Add new student record into existing data list structure.
Save updated student list back into the JSON file.
Search student record using roll number from stored data.
Update student marks if matching record is found successfully.
End the program after performing all required operations.

import json 
 
FILENAME = "students.json" 
 
def load_data(): 
    try: 
        with open(FILENAME, "r") as f: 
            return json.load(f) 
    except FileNotFoundError: 
        return []   
 
def save_data(data): 
    with open(FILENAME, "w") as f: 
        json.dump(data, f, indent=4) 
 
 
def add_student(roll_no, name, marks): 
    data = load_data() 
    student = {"roll_no": roll_no, "name": name, "marks": marks} 
    data.append(student) 
    save_data(data) 
    print(f"Student {name} added successfully.") 
 
 
def search_student(roll_no): 
    data = load_data() 
    for student in data: 
        if student["roll_no"] == roll_no: 
            print("Student Found:", student) 
            return student 
    print("Student not found.") 
    return None 
 
 
def update_marks(roll_no, new_marks): 
    data = load_data() 
    for student in data: 
        if student["roll_no"] == roll_no: 
            student["marks"] = new_marks 
            save_data(data) 
            print(f"Marks updated for {student['name']}.") 
            return 
    print("Student not found.") 
 
if __name__ == "__main__": 
    add_student(101, "Alice", 85) 
    add_student(102, "Bob", 90) 
    add_student(103, "Charlie", 78) 
    search_student(102) 
    update_marks(103, 88) 
    search_student(103)
