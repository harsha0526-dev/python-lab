from functools import reduce 
 
 
def factorial_recursive(n): 
    if n == 0 or n == 1: 
        return 1 
    else: 
        return n * factorial_recursive(n - 1) 
 
 
factorial_lambda = lambda n: reduce(lambda x, y: x * y, range(1, n + 1), 1) 
 
 
def main(): 
    try: 
        num = int(input("Enter a number to compute factorial: ")) 
 
        print(f"Factorial of {num} using recursion: {factorial_recursive(num)}") 
 
        print(f"Factorial of {num} using lambda & reduce: {factorial_lambda(num)}") 
 
    except ValueError: 
        print("Invalid input! Please enter an integer.") 
 
 
if __name__ == "__main__": 
    main()
