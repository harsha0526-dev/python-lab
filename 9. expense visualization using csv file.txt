import pandas as pd
import matplotlib.pyplot as plt

data = pd.read_csv(r"C:\Users\Admin\Documents\expenses.csv")


plt.figure(figsize=(10, 5))
plt.plot(data['Month'], data['Expense'], marker='o', linestyle='-', color='b', label='Expenses')
plt.title("Monthly Expenses - Line Chart")
plt.xlabel("Month")
plt.ylabel("Expense Amount")
plt.legend()
plt.grid(True)
plt.show()

plt.figure(figsize=(10, 5))
plt.bar(data['Month'], data['Expense'], color='orange', label='Expenses')
plt.title("Monthly Expenses - Bar Chart")
plt.xlabel("Month")
plt.ylabel("Expense Amount")
plt.legend()
plt.show()
