11. Tkinter Calculator
Start the program and import tkinter and messagebox modules.
Create main application window with fixed size and title.
Add labels and entry fields to accept user input values.
Define calculate function to perform arithmetic operations.
Retrieve input values and convert them into float type.
Perform operation based on button pressed by user.
Handle division by zero using exception handling mechanism.
Display calculated result using label widget in window.
Handle invalid input using error message box display.
Run main event loop to display graphical interface continuously.

import tkinter as tk
from tkinter import messagebox

def calculate(operation):
    try:
        num1 = float(entry1.get())
        num2 = float(entry2.get())

        if operation == "add":
            result = num1 + num2
        elif operation == "subtract":
            result = num1 - num2
        elif operation == "multiply":
            result = num1 * num2
        elif operation == "divide":
            if num2 == 0:
                raise ZeroDivisionError("Cannot divide by zero")
            result = num1 / num2
        else:
            result = "Invalid Operation"

        result_label.config(text=f"Result: {result}")
    except ValueError:
        messagebox.showerror("Input Error", "Please enter valid numbers.")
    except ZeroDivisionError as e:
        messagebox.showerror("Math Error", str(e))


# Create main window
root = tk.Tk()
root.title("Simple Calculator")
root.geometry("300x250")
root.resizable(False, False)

# Labels and Entry fields
tk.Label(root, text="Enter first number:").pack(pady=5)
entry1 = tk.Entry(root, width=20)
entry1.pack(pady=5)

tk.Label(root, text="Enter second number:").pack(pady=5)
entry2 = tk.Entry(root, width=20)
entry2.pack(pady=5)

# Buttons for operations
frame = tk.Frame(root)
frame.pack(pady=10)

btn_add = tk.Button(frame, text="Add", width=10, command=lambda: calculate("add"))
btn_add.grid(row=0, column=0, padx=5, pady=5)

btn_sub = tk.Button(frame, text="Subtract", width=10, command=lambda: calculate("subtract"))
btn_sub.grid(row=0, column=1, padx=5, pady=5)

btn_mul = tk.Button(frame, text="Multiply", width=10, command=lambda: calculate("multiply"))
btn_mul.grid(row=1, column=0, padx=5, pady=5)

btn_div = tk.Button(frame, text="Divide", width=10, command=lambda: calculate("divide"))
btn_div.grid(row=1, column=1, padx=5, pady=5)

# Result label
result_label = tk.Label(root, text="Result: ", font=("Arial", 12))
result_label.pack(pady=15)

# Run the application
root.mainloop()
