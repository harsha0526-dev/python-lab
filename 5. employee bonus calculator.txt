class Employee:
    def __init__(self, emp_id, name, base_salary, performance_rating):
        """
        Initialize employee attributes.
        :param emp_id: Unique identifier for the employee
        :param name: Employee's name
        :param base_salary: Employee's base salary
        :param performance_rating: Performance rating (scale 1–5)
        """
        self.emp_id = emp_id
        self.name = name
        self.base_salary = base_salary
        self.performance_rating = performance_rating

    def calculate_bonus(self):
        """
        Calculate performance-based bonus using conditional logic.
        Bonus percentage depends on performance rating:
        - 5 → 20% of base salary
        - 4 → 15% of base salary
        - 3 → 10% of base salary
        - 2 → 5% of base salary
        - 1 → No bonus
        """
        if self.performance_rating == 5:
            bonus_percentage = 0.20
        elif self.performance_rating == 4:
            bonus_percentage = 0.15
        elif self.performance_rating == 3:
            bonus_percentage = 0.10
        elif self.performance_rating == 2:
            bonus_percentage = 0.05
        else:
            bonus_percentage = 0.0

        return self.base_salary * bonus_percentage

    def display_employee_info(self):
        """
        Display employee details along with calculated bonus.
        """
        bonus = self.calculate_bonus()
        print(f"Employee ID: {self.emp_id}")
        print(f"Name: {self.name}")
        print(f"Base Salary: {self.base_salary}")
        print(f"Performance Rating: {self.performance_rating}")
        print(f"Bonus: {bonus}")


emp1 = Employee(101, "Alice", 50000, 5)
emp2 = Employee(102, "Bob", 40000, 3)
emp3 = Employee(103, "Charlie", 35000, 1)

    # Display employee info with bonuses
emp1.display_employee_info()
print("------")

print("------")
